135     /**
136      * Sends the specified field/value pairs.
137      * 
138      * @param fields the fields.
139      * @param values the values.
140      */
141     @SuppressWarnings("unchecked")
142     private void sendFieldsWithValues(String[] fields, String[] values) {
143 	if (fields.length != values.length) {
144 	    throw new IllegalArgumentException(
145 		    "Mismatched Field and Value Sets fields.length != values.length");
146 	}
147 	if (this.workitem == null) {
148 	    throw new IllegalStateException("Workitem has not been set yet");
149 	}
150 
151 	for (int i = 0; i < fields.length; i++) {
152 	    this.workitem.put(fields[i], values[i]);
153 	}
154 
155 	BlockingConnection connection = null;
156 
157 	try {
158 	    connection = this.getConnection();
159 	    this.sendCurrentWorkitem(connection);
160 	} catch (IOException e) {
161 	    throw new RuntimeException("Error connecting to Stomp server", e);
162 	} catch (URISyntaxException e) {
163 	    throw new RuntimeException("Error connecting to Stomp server", e);
164 	} finally {
165 	    closeConnectionAndCleanup(connection, fields);
166 	}
167     }

169     /**
170      * Sends the current work item data to the server.
171      * 
172      * @param connection
173      * @throws IOException
174      */
175     private void sendCurrentWorkitem(BlockingConnection connection)
176 	    throws IOException {
177 
178 	Object queue = this.stompConfig.get("queue");
179 	if (queue == null) {
180 	    logger.log(Level.SEVERE,
181 		    "Missing Stomp Configuration. Make Sure Queue is Set");
182 	    return;
183 	}
184 
185 	StompFrame frame = new StompFrame(SEND);
186 	frame.addHeader(DESTINATION, StompFrame.encodeHeader(queue.toString()));
187 	Buffer buffer = new Buffer(this.workitem.toJSONString().getBytes());
188 	frame.content(buffer);
189 
190 	connection.send(frame);
191 
192 	try {
193 	    Thread.sleep(500);
194 	} catch (InterruptedException ex) {
195 	    logger.log(Level.SEVERE, "Sleep interrupted", ex);
196 	}
197 
198     }

200     /**
201      * Gets a new stomp connection.
202      * 
203      * @return a stomp connection
204      * @throws IOException if the connection could not be established.
205      * @throws URISyntaxException if the host or port is missing from the
206      *                            stomp configuration.
207      */
208     private BlockingConnection getConnection() throws IOException,
209 	    URISyntaxException {
210 
211 	Object h = this.stompConfig.get("host");
212 	Object p = this.stompConfig.get("port");
213 
214 	if ((h == null) || (p == null)) {
215 	    throw new IllegalStateException(
216 		    "Missing Stomp Configuration. Make Sure Host and Port Are Set");
217 	}
218 
219 	return stompConnectionFactory.getConnection(h.toString(),
220 		Integer.parseInt(p.toString()));
221 
222     }

224     /**
225      * Closes the specified connection and cleans up the state of the worker.
226      * 
227      * @param connection the connection to close.
228      */
229     private void closeConnectionAndCleanup(BlockingConnection connection) {
230 	if (connection != null) {
231 	    connection.suspend();
232 	    try {
233 		connection.close();
234 	    } catch (IOException e) {
235 		// ignore
236 	    }
237 	}
238 
239 	this.workitem.remove(OUTPUT_META);
240 	this.workitem.remove(STREAMING_META);
241 	this.workitem.remove(CANCEL_META);
242 	if (this.workitem.get(WAITING_META) != null
243 		&& !Boolean.parseBoolean(this.workitem.get(WAITING_META).toString())) {
244 	    this.workitem.remove(WAITING_META);
245 	}
246     }

248     /**
249      * Closes the stomp connection and cleans up.
250      * 
251      * @param connection the stomp connection.
252      * @param fields the work item fields.
253      */
254     private void closeConnectionAndCleanup(BlockingConnection connection,
255 	    String[] fields) {
256 	this.closeConnectionAndCleanup(connection);
257 	for (String field : fields) {
258 	    if (!WAITING_META.equals(field))
259 		this.workitem.remove(field);
260 	}
261     }

294      * Returns the specified array field from the work item.
295      * 
296      * Caveat: lists of integers are returned as lists of longs
297      * 
298      * @param clazz the type of object contained in the field's array.
299      * @param field key to get value for
300      * @return a list containing the field's values.
301      */
302     @SuppressWarnings("unchecked")
303     public <T> List<T> getArrayField(Class<T> clazz, String field) {
304 	JSONObject fields = getFields();
305 	Object o = fields.get(field);
306 	if (o == null) {
307 	    return null;
308 	}
309 	if (o instanceof List) {
310 	    return (List<T>) o;
311 	}
312 	throw new IllegalArgumentException(format(
313 		"Field %s is not an array nor can be parsed as such: %s",
314 		field, o));
315     }

317     /**
318      * Main entry point from the agent.
319      * 
320      * @param methodName name of the method to execute.
321      * @param workitem the work item.
322      * @return the work item
323      */
324     @SuppressWarnings("rawtypes")
325     public final Map perform(String methodName, Map workitem) {
326 	String className = this.getClass().getName();
327 	try {
328 	    JSONParser parser = new JSONParser();
329 	    String json = JSONObject.toJSONString(workitem);
330 	    setWorkitem((JSONObject) parser.parse(json));
331 
332 	    writeOutput(format("Executing plugin: %s.%s%n", className, methodName));
333 
334 	    Method method = getClass().getMethod(methodName);
335 	    method.invoke(this);
336 
337 	    writeOutput(format("Finished plugin execution: %s.%s%n", className,
338 		    methodName));
339 
340 	} catch (InvocationTargetException e) {
341 	    // get the root cause of the exception
342 	    String msg = format("Plugin %s.%s failed: %s ", className, methodName,
343 		    getStackTrace(e.getCause()));
344 	    this.writeOutput(msg);
345 	    this.setError(msg);
346 	} catch (Exception e) {
347 	    String msg = format("Plugin %s.%s failed: %s ", className, methodName,
348 		    getStackTrace(e));
349 	    this.writeOutput(msg);
350 	    this.setError(msg);
351 	}
352 	return getWorkitem();
353     }

469     void createRecordWithFields(String model, String[] recordFields,

480 	    logger.log(Level.SEVERE, "Error creating record, fields: "

482 		    + StringUtils.join(recordValues, ","), e);
